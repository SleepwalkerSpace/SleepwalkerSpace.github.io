import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

// MyApp 根;
class MyApp extends StatelessWidget {
  
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(
          title: Text("flutter"),  
        ),
        body: Layout(),
      ),
    );
  }

}


class Layout extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      height: 500,
      child: Row(
        // 均匀分配;
        mainAxisAlignment: MainAxisAlignment.spaceEvenly,
        crossAxisAlignment: CrossAxisAlignment.end,
        children: <Widget>[
          IconContainer(Icons.home, size: 64,color: Colors.green[200]),
          IconContainer(Icons.search, size: 64,color: Colors.red[200]),
          IconContainer(Icons.book, size: 64,color: Colors.blue[200]),
          IconContainer(Icons.call, size: 64,color: Colors.yellow[200]),
        ],
      ),
    );
  }
  
}

class IconContainer extends StatelessWidget {

  IconData icon;
  double size = 32.0;
  Color color = Colors.red[300];


  IconContainer(this.icon, {this.size ,this.color}){

  }
  
  @override
  Widget build(Object context) {
    return Container(
      width: 100.0,
      height: 100.0,
      color: Colors.pink[50],
      child: Center(
        child: Icon(this.icon, size: this.size, color: this.color),
      ),
    );
  }
  
}

class Layout extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Row(
        children: <Widget>[
          Expanded(
            flex: 1,
            child: IconContainer(Icons.home, size: 64,color: Colors.green[100]),
          ),
          Expanded(
            flex: 2,
            child: IconContainer(Icons.home, size: 64,color: Colors.green[200]),
          ),
          Expanded(
            flex: 1,
            child: IconContainer(Icons.call, size: 64,color: Colors.green[300]),
          ),
        ]
      );
  }
  
}
